// 컴퓨터 0 ~ 9 중복 X 숫자 무작위 설정
void Set_com_nums(int com[3], int len)
{
	for (int i = 0; i < len; ++i)
	{
		com[i] = rand() % 10;
		for (int j = 0; j < i; ++j)
		{
			if (com[i] == com[j])
			{
				--i;
				break;
			}
		}
	}
}
