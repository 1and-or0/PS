#include <stdio.h>
#include <stdlib.h>

void test(int data[5], int *b)
{
	for (int i = 0; i < 5; ++i)
	{
		data[i] = data[i] * 2;
	}
	*b = 100;
	printf("my[0] = b 입력: ");
	scanf_s("%d", b);
}


int main()
{
	int my[5] = {1, 2, 3, 4, 5};

	test(my, &my[0]);

	for (int i = 0; i < 5; ++i)
	{
		printf("%d\n", my[i]);
	}

	return 0;
}
